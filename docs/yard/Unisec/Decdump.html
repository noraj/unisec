def self.utf16be(str)
  dec_chuncks = str.encode('UTF-16BE').to_hex.scan(/.{2}/).map do |x|
    x.hex2dec(padding: 3)
  end
  dec_chuncks.join(' ').scan(/\d+ \d+/).map { |x| "|#{x}|" }.join(' ')
end

def display

  "UTF-16BE: #{@utf16be}\n" \
  "UTF-16LE: #{@utf16le}\n" \
  "UTF-32BE: #{@utf32be}\n" \
  "UTF-32LE: #{@utf32le}".gsub('|', Paint['|', :red])
